```vue
<!-- src/views/inventory/ExpiryWarningList.vue -->
<template>
  <div class="expiry-warning-list">
    <el-card>
      <!-- 搜索栏 -->
      <el-form :inline="true" :model="searchForm" class="search-form">
        <el-form-item label="预警天数">
          <el-select v-model="searchForm.days" placeholder="请选择">
            <el-option label="7天内" :value="7" />
            <el-option label="30天内" :value="30" />
            <el-option label="90天内" :value="90" />
          </el-select>
        </el-form-item>
        <el-form-item label="物品类型">
          <el-select v-model="searchForm.itemType" placeholder="全部" clearable>
            <el-option label="物料" value="material" />
            <el-option label="制剂" value="preparation" />
          </el-select>
        </el-form-item>
        <el-form-item label="预警级别">
          <el-select v-model="searchForm.warningLevel" placeholder="全部" clearable>
            <el-option label="紧急" value="urgent" />
            <el-option label="警告" value="warning" />
            <el-option label="提示" value="info" />
          </el-select>
        </el-form-item>
        <el-form-item>
          <el-button type="primary" @click="handleSearch">查询</el-button>
          <el-button @click="handleReset">重置</el-button>
        </el-form-item>
      </el-form>

      <!-- 数据表格 -->
      <el-table :data="tableData" border stripe v-loading="loading">
        <el-table-column prop="itemCode" label="物品编码" width="120" />
        <el-table-column prop="itemName" label="物品名称" width="150" />
        <el-table-column prop="batchNumber" label="批次号" width="150" />
        <el-table-column prop="quantity" label="库存数量" width="100" />
        <el-table-column prop="expiryDate" label="有效期至" width="120">
          <template slot-scope="scope">
            {{ scope.row.expiryDate }}
          </template>
        </el-table-column>
        <el-table-column prop="remainingDays" label="剩余天数" width="100">
          <template slot-scope="scope">
            <el-tag :type="getRemainingDaysType(scope.row.remainingDays)">
              {{ scope.row.remainingDays }}天
            </el-tag>
          </template>
        </el-table-column>
        <el-table-column prop="warningLevel" label="预警级别" width="100">
          <template slot-scope="scope">
            <el-tag :type="getWarningLevelType(scope.row.warningLevel)">
              {{ getWarningLevelText(scope.row.warningLevel) }}
            </el-tag>
          </template>
        </el-table-column>
        <el-table-column prop="firstInDate" label="首次入库时间" width="160">
          <template slot-scope="scope">
            {{ scope.row.firstInDate ? formatDateTime(scope.row.firstInDate) : '-' }}
          </template>
        </el-table-column>
        <el-table-column prop="storageLocation" label="库位" width="100" />
        <el-table-column label="操作" width="100" fixed="right">
          <template slot-scope="scope">
            <el-button type="text" size="small" @click="handleViewDetail(scope.row)">
              详情
            </el-button>
          </template>
        </el-table-column>
      </el-table>

      <!-- 分页 -->
      <el-pagination
        @current-change="handlePageChange"
        @size-change="handleSizeChange"
        :current-page="pagination.pageIndex + 1"
        :page-sizes="[10, 20, 50, 100]"
        :page-size="pagination.pageSize"
        layout="total, sizes, prev, pager, next, jumper"
        :total="pagination.totalCount"
        style="margin-top: 20px; text-align: right"
      />
    </el-card>
  </div>
</template>

<script>
import { queryExpiringStocks } from '@/api/expiryWarning';
import dayjs from 'dayjs';

export default {
  name: 'ExpiryWarningList',
  data() {
    return {
      loading: false,
      searchForm: {
        days: 90,
        itemType: '',
        prodUnitId: null,
        warningLevel: ''
      },
      tableData: [],
      pagination: {
        pageIndex: 0,
        pageSize: 20,
        totalCount: 0
      }
    };
  },
  created() {
    this.loadData();
  },
  methods: {
    async loadData() {
      this.loading = true;
      try {
        const params = {
          ...this.searchForm,
          pageIndex: this.pagination.pageIndex,
          pageSize: this.pagination.pageSize
        };
        const res = await queryExpiringStocks(params);
        if (res.code === 200) {
          this.tableData = res.data.items;
          this.pagination.totalCount = res.data.totalCount;
        }
      } catch (error) {
        this.$message.error('加载数据失败');
      } finally {
        this.loading = false;
      }
    },
    handleSearch() {
      this.pagination.pageIndex = 0;
      this.loadData();
    },
    handleReset() {
      this.searchForm = {
        days: 90,
        itemType: '',
        prodUnitId: null,
        warningLevel: ''
      };
      this.handleSearch();
    },
    handlePageChange(page) {
      this.pagination.pageIndex = page - 1;
      this.loadData();
    },
    handleSizeChange(size) {
      this.pagination.pageSize = size;
      this.pagination.pageIndex = 0;
      this.loadData();
    },
    handleViewDetail(row) {
      // 跳转到库存详情页或显示对话框
      this.$message.info(`查看 ${row.itemName} 的详细信息`);
    },
    getRemainingDaysType(days) {
      if (days <= 7) return 'danger';
      if (days <= 30) return 'warning';
      return 'info';
    },
    getWarningLevelType(level) {
      const types = {
        urgent: 'danger',
        warning: 'warning',
        info: 'info'
      };
      return types[level] || 'info';
    },
    getWarningLevelText(level) {
      const texts = {
        urgent: '紧急',
        warning: '警告',
        info: '提示'
      };
      return texts[level] || level;
    },
    formatDateTime(dateTime) {
      return dayjs(dateTime).format('YYYY-MM-DD HH:mm:ss');
    }
  }
};
</script>

<style scoped>

  margin-bottom: 20px;
}
</style>
```
